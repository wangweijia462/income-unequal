% ======================== 理论机制 ========================
\begin{frame}{理论机制与研究假说}
  \begin{columns}[T,totalwidth=\textwidth]
    \begin{column}{0.52\textwidth}
      \small 在 Eaton \& Kortum (2002) 引力模型与 Melitz (2003) 异质企业框架的基础上，将数字限制视为\alert{数字关税等价物 (digital tariff equivalents)}：
      \begin{align*}
        \tau_{ij}(\theta_j,I_{ij}) &= \exp\{-\alpha\theta_j-\beta\theta_j^{2}-\delta I_{ij}\} \\[-0.2em]
        \ln X_{ij} &= \kappa_{ij}+\eta_1\theta_j+\eta_2\theta_j^{2}+\eta_3 I_{ij}+\varepsilon_{ij}
      \end{align*}
      \footnotesize
      \begin{itemize}
        \item $\theta_j$：开放度（DSTOI），$\eta_1>0,\eta_2<0$ 体现非线性边际效应；
        \item $I_{ij}$：是否处于 CPTPP 协调框架内；
        \item 监管限制 → 提高可变贸易成本与企业进入合规成本；
        \item 协定 → 降低制度异质性、减少“数字关税”。
      \end{itemize}
    \end{column}
    \begin{column}{0.43\textwidth}
      \begin{block}{H1：开放度直接促出口}
        \footnotesize 提高 DSTOI 显著促进数字服务出口，并可能表现出非线性。
      \end{block}
      \begin{block}{H2：CPTPP 提高开放水平}
        \footnotesize 加入 CPTPP 降低成员国贸易监管程度，提高 DSTOI。
      \end{block}
      \begin{block}{H3：CPTPP 促进基础设施}
        \footnotesize 加入 CPTPP 提高数字硬件基础设施（固定宽带订阅率）。
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

% ======================== DML 方法 ========================
\begin{frame}{双重机器学习识别策略}
  \begin{columns}[T,totalwidth=\textwidth]
    \begin{column}{0.50\textwidth}
      \small
      \textbf{部分线性模型}（PLR，Chernozhukov 等 2018）
      \begin{align*}
        Y_{it} &= \theta\, D_{it} + g(X_{it}) + \mu_i + \lambda_t + \varepsilon_{it}\\[-0.2em]
        D_{it} &= m(X_{it}) + \mu'_i + \lambda'_t + v_{it}
      \end{align*}
      \footnotesize
      $Y_{it}=\ln Export$；$D_{it}=$ DSTOI 或 CPTPP$\times$DSTOI；$X_{it}=$ 高维控制变量。
      \vspace{0.3em}

      \textbf{Neyman 正交得分}：
      \[
      \hat{\theta}=\frac{\sum (Y-\hat l(X))(D-\hat m(X))}{\sum (D-\hat m(X))^{2}}
      \]
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{block}{为什么选 DML}
        \footnotesize
        \begin{itemize}
          \item 用 ML 估计 $g(\cdot)$、$m(\cdot)$，自动捕捉非线性；
          \item 残差化后在“无法被 X 解释的部分”估计 $\theta$；
          \item \hl{交叉拟合（K-fold）}降低过拟合与正则化偏差；
          \item 保留传统估计量的 $\sqrt{n}$-一致性与渐近正态性；
          \item 可对比多种 ML：RF / Lasso / GBoost / Nnet。
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{附录 B：DML 算法步骤}
  \small
  \begin{enumerate}
    \item 将样本随机划分为 $K$ 个折，每次留一折作为验证集 $I_k$，其余作为训练集 $I_{-k}$；
    \item 在训练集上分别用 ML 估计：
      \begin{itemize}\footnotesize
        \item $\hat l(X)=\mathbb E[Y\mid X]$（ML：RF / Lasso / GBoost / Nnet）；
        \item $\hat m(X)=\mathbb E[D\mid X]$；
      \end{itemize}
    \item 计算验证集残差：$\tilde Y = Y - \hat l(X),\quad \tilde D = D - \hat m(X)$；
    \item 用残差对残差回归得到 $\hat\theta=\sum \tilde Y\tilde D / \sum \tilde D^2$；
    \item Neyman 正交得分使 $\hat\theta$ 对 $\hat l,\hat m$ 的一阶扰动\alert{鲁棒}（一阶矩为零）；
    \item 重复 $K$ 次（cross-fitting），取平均；
    \item 渐近方差由正交得分函数计算，$\sqrt{n}(\hat\theta-\theta)\xrightarrow{d}\mathcal N(0,\sigma^2)$。
  \end{enumerate}
\end{frame}
